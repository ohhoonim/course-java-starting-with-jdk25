```java
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.core.userdetails.User;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.security.core.userdetails.UserDetailsService;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.provisioning.InMemoryUserDetailsManager;
import org.springframework.security.web.SecurityFilterChain;

@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        http
            .csrf(csrf -> csrf.disable()) 
            
            // 1. 인가 (Authorization) 설정: 
            // 이 설정이 내부적으로 AuthorizationFilter와 AuthorizationManager를 구성합니다.
            .authorizeHttpRequests(authorize -> authorize
                // A. 권한 검사 규칙 설정
                .requestMatchers("/public/**", "/login", "/").permitAll()
                .requestMatchers("/admin/**").hasRole("ADMIN")
                .requestMatchers("/user/**").hasRole("USER")
                .anyRequest().authenticated()
            )
            // 2. 폼 로그인 및 로그아웃 설정
            .formLogin(form -> form.permitAll())
            .logout(logout -> logout.logoutSuccessUrl("/public/home").permitAll());

        return http.build();
    }
    
    // --- In-Memory UserDetailsService (인증을 위한 설정) ---
    
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }
    
    @Bean
    public UserDetailsService userDetailsService(PasswordEncoder passwordEncoder) {
        UserDetails user = User.builder()
            .username("user")
            .password(passwordEncoder.encode("userpass"))
            .roles("USER")
            .build();

        UserDetails admin = User.builder()
            .username("admin")
            .password(passwordEncoder.encode("adminpass"))
            .roles("ADMIN", "USER")
            .build();

        return new InMemoryUserDetailsManager(user, admin);
    }
}
```

```java
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class TestController {

    @GetMapping("/")
    public String home() {
        return "Welcome Home (Public access)";
    }

    @GetMapping("/public/info")
    public String publicInfo() {
        return "This is public information.";
    }

    @GetMapping("/user/dashboard")
    public String userDashboard() {
        return "User Dashboard: Access Granted for ROLE_USER.";
    }

    @GetMapping("/admin/settings")
    public String adminSettings() {
        return "Admin Settings: Access Granted for ROLE_ADMIN.";
    }
    
    @GetMapping("/data/protected")
    public String protectedData() {
        // hasRole('USER')나 hasRole('ADMIN')에 명시되지 않은 경로입니다.
        // anyRequest().authenticated() 규칙에 따라 로그인만 하면 접근 가능합니다.
        return "Protected Data: Only requires successful login.";
    }
}
```
