### 실습 2) unit test 코드 작성하기 

- 다음의 클래스를 테스트하는 unit test 코드를 아래와 같이 같이 작성하였습니다. 왜 3개의 단위 테스트를 작성했는지 이유를 조사해보시오. (AI 챗 활용)
- 블랙박스 테스트 설계 기법에 대해 조사하시오. 
- 화이트박스 테스트 설계 기법에 대해 조사하시오. 
- 유닛 테스트에 적절한 테스트 설계기법이 무엇일지 논리적으로 설명하시오.

```java
public class Calculator {
    public int divide(int a, int b) {
        return a / b;
    }
}
```


```java
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class CalculatorTest {

    // 1. 일반적인 양수 나눗셈 테스트
    @Test
    void testStandardDivision() {
        /*
         * 1. Arrange (준비)
         * - 테스트에 필요한 객체와 입력 데이터 설정
         */
        Calculator calculator = new Calculator();
        int dividend = 10;
        int divisor = 2;
        int expectedResult = 5;

        /*
         * 2. Act (실행)
         * - 테스트 대상 메소드를 실행
         */
        int actualResult = calculator.divide(dividend, divisor);

        /*
         * 3. Assert (검증)
         * - 실제 결과와 예상 결과 비교
         */
        assertEquals(expectedResult, actualResult, "10 나누기 2는 5여야 합니다.");
    }


    // 2. 나머지가 있는 나눗셈 테스트 (Java 정수 나눗셈의 특징 검증)
    @Test
    void testIntegerDivisionWithRemainder() {
        // Arrange
        Calculator calculator = new Calculator();
        int dividend = 10;
        int divisor = 3;
        // Java의 정수 나눗셈은 소수점 이하를 버림(truncate)
        int expectedResult = 3; 

        // Act
        int actualResult = calculator.divide(dividend, divisor);

        // Assert
        assertEquals(expectedResult, actualResult, "10 나누기 3은 정수 나눗셈이므로 3이어야 합니다.");
    }
    
    // 3. 0으로 나누는 예외 상황 테스트 (중요!)
    @Test
    void testDivisionByZeroShouldThrowException() {
        /*
         * 1. Arrange (준비)
         * - 테스트에 필요한 객체 설정
         */
        Calculator calculator = new Calculator();
        int dividend = 10;
        int divisor = 0; // 0으로 나누기 시도
        
        /*
         * 2. Act & Assert (실행 및 검증)
         * - 0으로 나눌 때 ArithmeticException이 발생하는지 검증
         * JUnit의 assertThrows를 사용하면 Act와 Assert를 동시에 수행합니다.
         */
        
        // Assert: 특정 예외(ArithmeticException.class)가 발생하는지 검증
        assertThrows(ArithmeticException.class, () -> {
            calculator.divide(dividend, divisor); // Act: 이 코드를 실행
        }, "0으로 나눌 때는 ArithmeticException이 발생해야 합니다.");
    }
}
```
