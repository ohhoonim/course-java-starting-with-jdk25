# jsonPath 라이브러리

## 실습

### 실습 1) JSONPath 연습 

- 아래 JSON 데이터를 이용하여 JSONPath를 연습하시오. 

```json
String storeDataJson = """
{
  "stores": [
    {
      "name": "Central Store",
      "address": "123 Main St",
      "inventory": {
        "book": [
          { "title": "The Lord of the Rings", "category": "fiction", "price": 22.99, "inStock": true },
          { "title": "Clean Code", "category": "programming", "price": 35.50, "inStock": false },
          { "title": "Moby Dick", "category": "fiction", "price": 8.99, "inStock": true }
        ],
        "bicycle": {
          "color": "red",
          "price": 19.95
        }
      }
    },
    {
      "name": "Branch Store",
      "address": "456 Side Ave",
      "inventory": {
        "book": [
          { "title": "Effective Java", "category": "programming", "price": 40.00, "inStock": true }
        ]
      }
    }
  ]
}
""";

```


```java

import com.jayway.jsonpath.JsonPath;
import java.util.List;
import java.util.Map;

// Jayway JSONPath 라이브러리를 사용한다고 가정합니다.

public class JsonPathPractice {

    public static void main(String[] args) {
        // (위의 storeDataJson 변수가 여기에 있다고 가정)
        // ... String storeDataJson = "...";

        // 1. 모든 상점의 이름(name) 추출
        String path1 = "$.stores[*].name";
        List<String> storeNames = JsonPath.read(storeDataJson, path1);
        System.out.println("1. 모든 상점 이름: " + storeNames);
        // 결과: [Central Store, Branch Store]

        // 2. 모든 책의 제목(title)을 재귀적 탐색(Deep Scan)으로 추출
        String path2 = "$..title";
        List<String> allTitles = JsonPath.read(storeDataJson, path2);
        System.out.println("2. 모든 책 제목: " + allTitles);
        // 결과: [The Lord of the Rings, Clean Code, Moby Dick, Effective Java]

        // 3. 가격이 25.00 미만인 책의 제목만 필터링하여 추출
        String path3 = "$.stores[*].inventory.book[?(@.price < 25.00)].title";
        List<String> cheapBookTitles = JsonPath.read(storeDataJson, path3);
        System.out.println("3. 25.00 미만 책 제목: " + cheapBookTitles);
        // 결과: [The Lord of the Rings, Moby Dick]

        // 4. "programming" 카테고리 책 중 재고(inStock)가 있는 책의 가격 추출
        String path4 = "$.stores[*].inventory.book[?(@.category == 'programming' && @.inStock == true)].price";
        List<Double> availablePrices = JsonPath.read(storeDataJson, path4);
        System.out.println("4. 재고 있는 프로그래밍 책 가격: " + availablePrices);
        // 결과: [40.0] (Branch Store의 Effective Java)
        
        // 5. 첫 번째 상점의 자전거(bicycle) 색상 추출
        String path5 = "$.stores[0].inventory.bicycle.color";
        String bicycleColor = JsonPath.read(storeDataJson, path5);
        System.out.println("5. 첫 번째 상점 자전거 색상: " + bicycleColor);
        // 결과: red
    }
}
```
