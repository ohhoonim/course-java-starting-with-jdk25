> 🗒️ Static Import(정적 임포트)란? 
> - Static Import (정적 임포트)는 Java 언어에서 제공하는 기능으로, 일반적인 클래스 임포트와 달리 특정 클래스의 static 멤버(메소드, 필드)를 호출할 때 클래스 이름 없이 바로 사용할 수 있게 해주는 기능입니다.
> - 일반적인 임포트가 클래스 자체를 가져오는 것이라면, 정적 임포트는 클래스 내부에 있는 **정적인 요소(static members)**만 가져오는 것입니다.

```java
// src/main/java/com/example/Book.java
public class Book {
    private final String title;
    private final String author;
    private final int pages;
    private final boolean isBestSeller;

    public Book(String title, String author, int pages, boolean isBestSeller) {
        this.title = title;
        this.author = author;
        this.pages = pages;
        this.isBestSeller = isBestSeller;
    }

    // Getter methods
    public String getTitle() { return title; }
    public String getAuthor() { return author; }
    public int getPages() { return pages; }
    public boolean isBestSeller() { return isBestSeller; }
}
```

```java
// src/test/java/com/example/BookTest.java
import org.junit.jupiter.api.Test;
import java.util.List;
import java.util.Arrays;

// AssertJ의 핵심 (static import)
import static org.assertj.core.api.Assertions.*;

public class BookTest {

    // 1. 객체 단언(Object Assertions) 및 논리값 검증
    @Test
    void testSingleBookProperties() {
        // Arrange (준비)
        Book javaBook = new Book("Effective Java", "Joshua Bloch", 688, true);

        // Act (실행) - (간단한 객체 검증에서는 Act 단계가 생략되기도 함)

        // Assert (검증) - AssertJ를 사용한 유창한 단언
        
        // 1. 기본 객체 검증 (null 여부)
        assertThat(javaBook).isNotNull();

        // 2. 문자열 속성 검증
        assertThat(javaBook.getTitle())
            .startsWith("Effective")
            .endsWith("Java")
            .containsIgnoringCase("java"); // 대소문자 무시 검증

        // 3. 숫자 속성 검증
        assertThat(javaBook.getPages())
            .isGreaterThan(600) // 600 페이지 초과인지
            .isLessThanOrEqualTo(700); // 700 페이지 이하인지
            
        // 4. 논리값 검증
        assertThat(javaBook.isBestSeller()).isTrue(); // True인지 검증
    }

    // 2. 컬렉션 단언(Collection Assertions)
    @Test
    void testBookCollection() {
        // Arrange (준비)
        List<String> authors = Arrays.asList("Joshua Bloch", "Kent Beck", "Erich Gamma");

        // Assert (검증)
        
        // 1. 크기 및 내용 검증
        assertThat(authors)
            .hasSize(3) // 리스트 크기가 3인지
            .contains("Kent Beck") // "Kent Beck"을 포함하는지
            .containsExactlyInAnyOrder("Erich Gamma", "Joshua Bloch", "Kent Beck") // 순서에 상관없이 정확히 이 요소들만 포함하는지
            .doesNotContain("Robert C. Martin"); // 특정 요소를 포함하지 않는지
            
        // 2. 모든 요소 검증 (Predicate 사용)
        assertThat(authors).allMatch(author -> author.length() > 5); // 모든 작가 이름이 5글자 초과인지
    }
    
    // 3. 예외 단언(Exception Assertions)
    @Test
    void testExceptionHandling() {
        // AssertJ의 assertThrows 대신 assertThatExceptionOfType을 사용한 예외 검증
        
        // Assert: ArithmeticException이 발생하는지 검증
        assertThatExceptionOfType(ArithmeticException.class)
            .isThrownBy(() -> {
                // Act: 0으로 나누는 실행 코드
                int result = 10 / 0; 
            })
            .withMessageContaining("zero"); // 예외 메시지에 "zero"가 포함되어 있는지 검증
    }

    // 4. 객체 필드 추출(Extracting Fields) 단언 (객체 리스트 검증)
    @Test
    void testExtractingFields() {
        // Arrange (준비)
        Book book1 = new Book("Clean Code", "Robert C. Martin", 464, true);
        Book book2 = new Book("The Pragmatic Programmer", "David Thomas", 352, false);
        List<Book> library = Arrays.asList(book1, book2);

        // Assert (검증)
        
        // 1. 리스트에서 'author' 필드만 추출하여 검증
        assertThat(library)
            .extracting("author") // 필드 이름을 문자열로 전달
            .contains("David Thomas", "Robert C. Martin");

        // 2. 리스트에서 'pages' 필드만 추출하여 평균값 등을 검증 (숫자 검증 확장)
        assertThat(library)
            .extracting(Book::getPages) // 메소드 참조를 사용
            .as("모든 책의 페이지 수")
            .containsOnlyOnce(464) // 464 페이지 책이 한 권만 있는지
            .allMatch(pages -> pages > 300); // 모든 책이 300페이지 초과인지
    }
}
```
