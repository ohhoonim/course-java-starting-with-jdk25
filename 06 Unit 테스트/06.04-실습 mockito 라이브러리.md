# mockito 라이브러리

## 실습

### 실습 1) Mockito 연습

- UserService 에 대한 테스트 코드를 작성하시오.

```java

// 1. 데이터 클래스
public class User {
    private final long id;
    private final String email;
    public User(long id, String email) {
        this.id = id;
        this.email = email;
    }
    public String getEmail() { return email; }
    // 기타 Getter, Setter 등 생략
}

// 2. 의존성 (인터페이스)
public interface UserRepository {
    User findById(long id);
    void save(User user);
}

// 3. 테스트 대상 (Service)
public class UserService {
    private final UserRepository userRepository;

    public UserService(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    public User getUserInfo(long id) {
        // Mocking을 통해 이 부분이 실제 DB에 접근하지 않도록 함
        return userRepository.findById(id); 
    }

    public void registerUser(long id, String email) {
        User newUser = new User(id, email);
        userRepository.save(newUser);
    }
}

```


```java
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import static org.mockito.Mockito.*;
import static org.junit.jupiter.api.Assertions.*;

@ExtendWith(MockitoExtension.class)
public class UserServiceTest {

    // 1. Arrange (준비) - Mock 객체 생성
    @Mock
    UserRepository mockRepository; 

    // 2. Arrange (준비) - 테스트 대상 객체에 Mock 객체 주입
    @InjectMocks
    UserService userService;

    // ----------------------------------------------------------------------
    // 시나리오 1: 상태 기반 테스트 (Stubbing을 이용한 값 반환 검증)
    // ----------------------------------------------------------------------
    @Test
    void testGetUserInfo_ShouldReturnUser() {
        // Arrange (준비)
        long userId = 10L;
        User expectedUser = new User(userId, "mock@test.com");
        
        // Stubbing: mockRepository.findById(10L)이 호출되면 expectedUser를 반환하도록 설정
        when(mockRepository.findById(userId)).thenReturn(expectedUser); 

        // Act (실행)
        User actualUser = userService.getUserInfo(userId);

        // Assert (검증)
        // 1. 상태 검증: 서비스가 반환한 객체가 Stubbing된 객체와 동일한지 확인
        assertNotNull(actualUser);
        assertEquals(expectedUser.getEmail(), actualUser.getEmail());
        
        // 2. 행위 검증: findById 메소드가 정확히 1번 호출되었는지 확인
        verify(mockRepository, times(1)).findById(userId); 
    }

    // ----------------------------------------------------------------------
    // 시나리오 2: 행위 기반 테스트 (저장 메소드 호출 검증)
    // ----------------------------------------------------------------------
    @Test
    void testRegisterUser_ShouldCallSaveOnce() {
        // Arrange (준비)
        long newId = 20L;
        String newEmail = "new@user.com";
        
        // * 저장 로직은 반환 값이 없으므로 Stubbing이 필요 없음

        // Act (실행)
        userService.registerUser(newId, newEmail);

        // Assert (검증)
        // Verification: save() 메소드가 정확히 1번 호출되었는지, 
        // 그리고 호출 시 인자가 newId와 newEmail을 가진 User 객체인지 검증
        
        // any() 대신 eq()를 사용하여 특정 값으로 생성된 객체가 전달되었는지 확인
        verify(mockRepository, times(1)).save(argThat(user -> 
            user.getEmail().equals(newEmail) && user.getId() == newId
        ));
    }
}
```
