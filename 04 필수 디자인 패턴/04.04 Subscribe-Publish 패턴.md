# 04.04 Subscribe-Publish 패턴

줄여서 Pub/Sub 패턴은 메시지를 주고받는 시스템의 발행자(Publisher)와 구독자(Subscriber)를 분리하는 비동기 메시징 패턴입니다. Observer 패턴의 변형으로 볼 수 있습니다.


- 구독: 구독자(Subscriber)는 브로커에게 특정 토픽(예: '날씨 정보')에 관심이 있음을 등록합니다.
- 발행: 발행자(Publisher)는 자신이 생성한 메시지를 토픽과 함께 브로커에게 보냅니다.
- 전달: 브로커는 해당 토픽을 구독한 모든 구독자에게 메시지를 복사하여 비동기적으로 전달합니다.

## Pub/Sub 패턴의 장점

- 느슨한 결합 (Decoupling): 발행자와 구독자가 서로의 존재를 모르기 때문에, 한쪽의 시스템이 변경되어도 다른 쪽에 영향을 주지 않아 시스템 확장이 매우 용이합니다 (유연성).
- 확장성 (Scalability): 구독자가 늘어나더라도 발행자 코드를 수정할 필요가 없습니다. 브로커만 부하를 처리할 수 있으면 됩니다.
- 비동기 처리: 메시지 발행 후 즉시 응답을 기다릴 필요 없이 다른 작업을 계속할 수 있어 처리 효율이 높습니다

## 주요 사용 사례

- 분산 시스템 및 마이크로 서비스: 서비스 간 데이터 통신 및 이벤트 전파 (Kafka, RabbitMQ, Redis Pub/Sub).
- 웹 개발: 실시간 알림, 채팅 기능, 상태 변화 감지.
- 운영체제 및 UI 이벤트: 사용자 액션(버튼 클릭)을 특정 컴포넌트(구독자)에게 전달하는 이벤트 시스템.

## 구현 예제

```java
// 이벤트 객체: 발행할 데이터(메시지)를 담고 있습니다.
class UserRegisteredEvent {
    private final String userId;
    private final String userEmail;

    UserRegisteredEvent(String userId, String userEmail) {
        this.userId = userId;
        this.userEmail = userEmail;
    }

    public String getUserId() {
        return userId;
    }

    public String getUserEmail() {
        return userEmail;
    }
}

// Publisher: 이벤트를 발행하는 클래스
class UserService {
    private final List<Object> eventListeners; // 구독자 목록 (Spring에서는 컨테이너가 관리)

    // 이 예제에서는 단순화를 위해 직접 리스너를 주입받는다고 가정
    UserService(List<Object> listeners) {
        this.eventListeners = listeners;
    }

    public void registerUser(String userId, String email) {
        IO.println("==================================================");
        IO.println("🚀 [Publisher: UserService] 사용자 등록 처리 중: " + userId);

        // 1. 핵심 비즈니스 로직 (사용자 DB 저장 등)

        // 2. 이벤트 객체 생성
        UserRegisteredEvent event = new UserRegisteredEvent(userId, email);

        // 3. 이벤트 발행 (구독자가 누가 있는지 알지 못함)
        publishEvent(event);

        IO.println("✅ [Publisher: UserService] 사용자 등록 및 이벤트 발행 완료.");
    }

    // 실제 Spring의 publishEvent() 역할 대행
    private void publishEvent(Object event) {
        // 이벤트가 발생하면, 모든 리스너에게 전달
        for (Object listener : eventListeners) {
            if (listener instanceof EmailService) {
                ((EmailService) listener).handleUserRegisteredEvent((UserRegisteredEvent) event);
            } else if (listener instanceof LoggingService) {
                ((LoggingService) listener).handleUserRegisteredEvent((UserRegisteredEvent) event);
            }
        }
    }
}

// Subscriber 1: 이메일 전송 로직 담당
class EmailService {
    // @EventListener 역할: 특정 이벤트 타입만 받아서 처리
    public void handleUserRegisteredEvent(UserRegisteredEvent event) {
        IO.println("🔔 [Subscriber: EmailService] " + event.getUserEmail() + "로 환영 이메일 전송 시작...");
        // 실제 이메일 전송 로직
        IO.println("   [EmailService] 이메일 전송 완료.");
    }
}

// Subscriber 2: 로깅 로직 담당
class LoggingService {
    // @EventListener 역할: 특정 이벤트 타입만 받아서 처리
    public void handleUserRegisteredEvent(UserRegisteredEvent event) {
        IO.println("📝 [Subscriber: LoggingService] 사용자 " + event.getUserId() + " 등록 이벤트 로그 기록.");
        // 실제 로깅 로직
    }
}

void main() {
    // 1. 구독자(리스너) 인스턴스 생성
    EmailService emailService = new EmailService();
    LoggingService loggingService = new LoggingService();

    // 2. 발행자에게 구독자 목록을 전달 (Spring 컨테이너 역할 대행)
    // Spring에서는 이 과정 없이 컨테이너가 자동으로 리스너를 찾아서 연결해줍니다.
    List<Object> listeners = Arrays.asList(emailService, loggingService);
    UserService userService = new UserService(listeners);

    // 3. 발행자에게 핵심 비즈니스 요청
    userService.registerUser("john_doe", "john@example.com");
}

```
