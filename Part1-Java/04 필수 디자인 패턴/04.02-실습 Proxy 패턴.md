# 04.02 Proxy 패턴

## 실습

### 실습 1) Spring framework와 AOP

- Proxy 패턴을 활용하면 기존 객체를 수정하지 않고도 기능을 추가할 수 있습니다.
- AOP(Aspect-Oriented Programming)에 대하여 조사하시오.
- Spring AOP의 두 가지 프록시 생성 기술을 조사하시오 

> 📋 Spring AOP의 두 가지 프록시 생성 기술
> - JDK Dynamic Proxy : 인터페이스를 구현하는 클래스를 대상 
> - CGLIB : 인터페이스 미구현 클래스를 대상
>
> CGLIB 방식인 경우 final 클래스는 proxy를 만들 수 없다

### 실습 2) Java의 java.lang.reflect.Proxy 클래스 

- Java의 Proxy 클래스를 이용하여 AOP를 구현하시오.
- https://docs.oracle.com/en/java/javase/25/docs/api//java.base/java/lang/reflect/Proxy.html

```java

// 1. Subject 인터페이스: 핵심 기능의 규약 (UserService)
interface UserService {
    String findUser(String userId);

    void saveUser(String userId, String name);
}

// 2. RealSubject: 핵심 비즈니스 로직을 가진 타겟 클래스
class RealUserService implements UserService {
    @Override
    public String findUser(String userId) {
        IO.println("   [Target] 사용자 데이터베이스를 조회 중입니다...");
        // 실제 데이터 조회 로직 (핵심 로직)
        if (userId.equals("admin")) {
            return "관리자 (ID: admin)";
        }
        return "일반 사용자 (ID: " + userId + ")";
    }

    @Override
    public void saveUser(String userId, String name) {
        IO.println("   [Target] 사용자 " + name + "의 정보를 DB에 저장 완료.");
    }
}

// AOP의 Aspect 역할을 수행하는 로깅 프록시 핸들러
class LoggingInvocationHandler implements InvocationHandler {
    private final Object target; // 실제 객체 (RealUserService)

    LoggingInvocationHandler(Object target) {
        this.target = target;
    }

    // 모든 메소드 호출을 가로채서 처리하는 핵심 메소드
    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {

        String methodName = method.getName();

        // 1. Before Advice (사전 로깅)
        IO.println("-> [Proxy] 메소드 시작: " + methodName + ", 인자: " + Arrays.toString(args));

        Object result = null;
        try {
            // 2. Target 호출 (핵심 비즈니스 로직 실행)
            result = method.invoke(target, args);

            // 3. After Returning Advice (정상 종료 후 로깅)
            IO.println("<- [Proxy] 메소드 종료: " + methodName + ", 반환값: " + result);
            return result;

        } catch (Throwable e) {
            // 4. After Throwing Advice (예외 발생 후 로깅)
            IO.println(
                    "<! [Proxy] 메소드 예외 발생: " + methodName + ", 오류: " + e.getCause().getMessage());
            throw e.getCause();
        }
    }

}

void main() {
    // 1. 실제 객체 (타겟) 생성
    UserService realService = new RealUserService();

    // 2. JDK Dynamic Proxy를 사용하여 프록시 객체 생성
    // 프록시 객체는 UserService 인터페이스를 구현하며, 모든 호출을 LoggingInvocationHandler로 보냅니다.
    UserService proxyService = (UserService) java.lang.reflect.Proxy.newProxyInstance(
            UserService.class.getClassLoader(), // 클래스 로더
            new Class<?>[] {UserService.class}, // 구현할 인터페이스
            new LoggingInvocationHandler(realService) // 모든 호출을 처리할 핸들러
    );

    IO.println("=================================================");
    IO.println("== [AOP 로깅 적용] findUser 메소드 호출 ==");

    // 3. 클라이언트는 프록시 객체를 사용 (실제 객체인 줄 압니다)
    String user = proxyService.findUser("guest_007");
    IO.println("최종 반환된 사용자: " + user);


    IO.println("\n=================================================");
    IO.println("== [AOP 로깅 적용] saveUser 메소드 호출 ==");

    proxyService.saveUser("newbie", "새로운 유저");

    IO.println("=================================================");
}

```
